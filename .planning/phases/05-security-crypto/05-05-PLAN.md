```go
// VaultKeyManager implements KeyManager using HashiCorp Vault transit engine.
// Runs on kubeadm K8s cluster as a Deployment in the vault namespace.
// Auth: Kubernetes SA (recommended) or Vault token.
type VaultKeyManager struct {
    address    string
    keyName    string
    authMethod string // "kubernetes" or "token"
    k8sRole    string
    token      string
    httpClient HTTPDoer
}

// NewVaultKeyManager creates a VaultKeyManager.
func NewVaultKeyManager(cfg *VaultConfig) *VaultKeyManager {
    return &VaultKeyManager{
        address:    cfg.Address,
        keyName:    cfg.KeyName,
        authMethod: cfg.AuthMethod,
        k8sRole:    cfg.K8sRole,
        token:      cfg.Token,
        httpClient: &http.Client{Timeout: 10 * time.Second},
    }
}

// vaultEncryptRequest is the request body for Vault transit /encrypt.
type vaultEncryptRequest struct {
    Plaintext string `json:"plaintext"` // base64-encoded DEK
}

// vaultEncryptResponse is the response body from Vault transit /encrypt.
type vaultEncryptResponse struct {
    Data struct {
        Ciphertext string `json:"ciphertext"` // Vault-managed ciphertext
    } `json:"data"`
}

// vaultDecryptRequest is the request body for Vault transit /decrypt.
type vaultDecryptRequest struct {
    Ciphertext string `json:"ciphertext"`
}

// vaultDecryptResponse is the response body from Vault transit /decrypt.
type vaultDecryptResponse struct {
    Data struct {
        Plaintext string `json:"plaintext"` // base64-encoded DEK
    } `json:"data"`
}

// vaultKeyInfo is the response from /v1/transit/keys/{key}.
type vaultKeyInfo struct {
    Data struct {
        Keys map[string]int `json:"keys"` // version -> creation_time
    } `json:"data"`
}

// Wrap encrypts a DEK using the Vault transit KEK.
// Calls POST /v1/transit/encrypt/<keyName>.
func (m *VaultKeyManager) Wrap(ctx context.Context, dek []byte) ([]byte, int, error) {
    // Encode DEK as base64 for Vault
    ptB64 := base64.StdEncoding.EncodeToString(dek)

    body, err := json.Marshal(vaultEncryptRequest{Plaintext: ptB64})
    if err != nil {
        return nil, 0, fmt.Errorf("vault wrap: marshal: %w", err)
    }

    url := fmt.Sprintf("%s/v1/transit/encrypt/%s", m.address, m.keyName)
    req, err := http.NewRequestWithContext(ctx, http.MethodPost, url, bytes.NewReader(body))
    if err != nil {
        return nil, 0, fmt.Errorf("vault wrap: new request: %w", err)
    }
    req.Header.Set("Content-Type", "application/json")

    if err := m.setAuthHeader(req); err != nil {
        return nil, 0, fmt.Errorf("vault wrap: auth: %w", err)
    }

    resp, err := m.httpClient.Do(req)
    if err != nil {
        return nil, 0, fmt.Errorf("vault wrap: request: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        bodyBytes, _ := io.ReadAll(resp.Body)
        return nil, 0, fmt.Errorf("vault wrap: status %d: %s", resp.StatusCode, string(bodyBytes))
    }

    var vaultResp vaultEncryptResponse
    if err := json.NewDecoder(resp.Body).Decode(&vaultResp); err != nil {
        return nil, 0, fmt.Errorf("vault wrap: decode: %w", err)
    }

    // Fetch current key version to return to caller
    ver, err := m.GetKeyVersion(ctx)
    if err != nil {
        ver = 1 // fallback to 1 if version fetch fails
    }
    return []byte(vaultResp.Data.Ciphertext), ver, nil
}

// Unwrap decrypts a wrapped DEK using the Vault transit KEK.
// Calls POST /v1/transit/decrypt/<keyName>.
func (m *VaultKeyManager) Unwrap(ctx context.Context, wrappedDEK []byte) ([]byte, error) {
    body, err := json.Marshal(vaultDecryptRequest{Ciphertext: string(wrappedDEK)})
    if err != nil {
        return nil, fmt.Errorf("vault unwrap: marshal: %w", err)
    }

    url := fmt.Sprintf("%s/v1/transit/decrypt/%s", m.address, m.keyName)
    req, err := http.NewRequestWithContext(ctx, http.MethodPost, url, bytes.NewReader(body))
    if err != nil {
        return nil, fmt.Errorf("vault unwrap: new request: %w", err)
    }
    req.Header.Set("Content-Type", "application/json")

    if err := m.setAuthHeader(req); err != nil {
        return nil, fmt.Errorf("vault unwrap: auth: %w", err)
    }

    resp, err := m.httpClient.Do(req)
    if err != nil {
        return nil, fmt.Errorf("vault unwrap: request: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        bodyBytes, _ := io.ReadAll(resp.Body)
        return nil, fmt.Errorf("vault unwrap: status %d: %s", resp.StatusCode, string(bodyBytes))
    }

    var vaultResp vaultDecryptResponse
    if err := json.NewDecoder(resp.Body).Decode(&vaultResp); err != nil {
        return nil, fmt.Errorf("vault unwrap: decode: %w", err)
    }

    dek, err := base64.StdEncoding.DecodeString(vaultResp.Data.Plaintext)
    if err != nil {
        return nil, fmt.Errorf("vault unwrap: decode base64: %w", err)
    }

    return dek, nil
}

// GetKeyVersion fetches the current key version from Vault.
// Calls GET /v1/transit/keys/<keyName>.
func (m *VaultKeyManager) GetKeyVersion(ctx context.Context) (int, error) {
    url := fmt.Sprintf("%s/v1/transit/keys/%s", m.address, m.keyName)
    req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
    if err != nil {
        return 0, fmt.Errorf("vault keyversion: new request: %w", err)
    }

    if err := m.setAuthHeader(req); err != nil {
        return 0, fmt.Errorf("vault keyversion: auth: %w", err)
    }

    resp, err := m.httpClient.Do(req)
    if err != nil {
        return 0, fmt.Errorf("vault keyversion: request: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return 0, fmt.Errorf("vault keyversion: status %d", resp.StatusCode)
    }

    var info vaultKeyInfo
    if err := json.NewDecoder(resp.Body).Decode(&info); err != nil {
        return 0, fmt.Errorf("vault keyversion: decode: %w", err)
    }

    maxVer := 0
    for v := range info.Data.Keys {
        var vInt int
        if _, err := fmt.Sscanf(v, "%d", &vInt); err != nil {
            continue
        }
        if vInt > maxVer {
            maxVer = vInt
        }
    }
    if maxVer == 0 {
        return 1, nil
    }
    return maxVer, nil
}

// RotateKey calls POST /v1/transit/rotate/<keyName> to generate a new KEK version.
func (m *VaultKeyManager) RotateKey(ctx context.Context) error {
    url := fmt.Sprintf("%s/v1/transit/rotate/%s", m.address, m.keyName)
    req, err := http.NewRequestWithContext(ctx, http.MethodPost, url, nil)
    if err != nil {
        return fmt.Errorf("vault rotate: new request: %w", err)
    }

    if err := m.setAuthHeader(req); err != nil {
        return fmt.Errorf("vault rotate: auth: %w", err)
    }

    resp, err := m.httpClient.Do(req)
    if err != nil {
        return fmt.Errorf("vault rotate: request: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusNoContent && resp.StatusCode != http.StatusOK {
        bodyBytes, _ := io.ReadAll(resp.Body)
        return fmt.Errorf("vault rotate: status %d: %s", resp.StatusCode, string(bodyBytes))
    }

    return nil
}

// setAuthHeader sets the appropriate Vault authentication header.
func (m *VaultKeyManager) setAuthHeader(req *http.Request) error {
    switch m.authMethod {
    case "kubernetes":
        // Read Kubernetes Service Account token from the mounted token file
        tokenBytes, err := os.ReadFile("/var/run/secrets/kubernetes.io/serviceaccount/token")
        if err != nil {
            return fmt.Errorf("read k8s SA token: %w", err)
        }
        req.Header.Set("X-Vault-Request", "true")
        req.Header.Set("Authorization", "Bearer "+string(tokenBytes))

    case "token":
        req.Header.Set("X-Vault-Request", "true")
        req.Header.Set("Authorization", "Bearer "+m.token)

    default:
        return errors.New("unsupported auth method: " + m.authMethod)
    }
    return nil
}
```

```go
//go:build softhsm

package crypto

import (
    "context"
    "errors"
    "fmt"

    "github.com/miekg/pkcs11"
)

// SoftHSMKeyManager implements KeyManager using SoftHSM2 via PKCS#11.
// For development and testing only. Requires libsofthsm2.so and initialized tokens.
// Gate this file with //go:build softhsm to avoid linking libsofthsm2.so in production builds.
type SoftHSMKeyManager struct {
    libraryPath string
    tokenLabel  string
    pin         string
    ctx         *pkcs11.Ctx
    session     pkcs11.SessionHandle
    kekObject   pkcs11.ObjectHandle
    keyID       []byte // CKA_ID for the KEK object
}

// NewSoftHSMKeyManager initializes the SoftHSM2 PKCS#11 context.
// Reads token from SOFTHSM2_CONF or default path.
func NewSoftHSMKeyManager(cfg *SoftHSMConfig) (*SoftHSMKeyManager, error) {
    libPath := cfg.LibraryPath
    if libPath == "" {
        libPath = "/usr/lib/softhsm/libsofthsm2.so"
    }
    ctx := pkcs11.New(libPath)
    if ctx == nil {
        return nil, errors.New("failed to initialize PKCS#11 context for SoftHSM2")
    }

    // Initialize the library
    if err := ctx.Initialize(); err != nil {
        return nil, fmt.Errorf("PKCS#11 initialize: %w", err)
    }

    // Open RO session on the slot with the matching token label
    var session pkcs11.SessionHandle
    var found bool
    slots, err := ctx.GetSlotList(true)
    if err != nil {
        return nil, fmt.Errorf("get slot list: %w", err)
    }
    for _, slot := range slots {
        tokenInfo, err := ctx.GetTokenInfo(slot)
        if err != nil {
            continue
        }
        if tokenInfo.Label == cfg.TokenLabel {
            session, err = ctx.OpenSession(slot, pkcs11.CKF_SERIAL_SESSION|pkcs11.CKF_RW_SESSION)
            if err != nil {
                return nil, fmt.Errorf("open session on slot %d: %w", slot, err)
            }
            found = true
            break
        }
    }
    if !found {
        return nil, fmt.Errorf("token with label %q not found in SoftHSM2", cfg.TokenLabel)
    }

    // Login (user PIN)
    if err := ctx.Login(session, pkcs11.CKU_USER, cfg.PIN); err != nil {
        return nil, fmt.Errorf("login: %w", err)
    }

    // Find the KEK object by label
    kekHandle, keyID, err := findKEKObject(ctx, session, cfg.TokenLabel)
    if err != nil {
        ctx.CloseSession(session)
        return nil, fmt.Errorf("find KEK object: %w", err)
    }

    return &SoftHSMKeyManager{
        libraryPath: libPath,
        tokenLabel: cfg.TokenLabel,
        pin:        cfg.PIN,
        ctx:        ctx,
        session:    session,
        kekObject:  kekHandle,
        keyID:      keyID,
    }, nil
}

// findKEKObject searches for the KEK object in the token by label.
func findKEKObject(ctx *pkcs11.Ctx, session pkcs11.SessionHandle, label string) (pkcs11.ObjectHandle, []byte, error) {
    // Search by CKA_LABEL
    template := []*pkcs11.Attribute{
        pkcs11.NewAttribute(pkcs11.CKA_LABEL, []byte(label)),
        pkcs11.NewAttribute(pkcs11.CKA_CLASS, pkcs11.CKO_SECRET_KEY),
        pkcs11.NewAttribute(pkcs11.CKA_KEY_TYPE, pkcs11.CKK_AES),
    }
    if err := ctx.FindObjectsInit(session, template); err != nil {
        return 0, nil, err
    }
    defer ctx.FindObjectsFinal(session)

    objects, _, err := ctx.FindObjects(session, 1)
    if err != nil {
        return 0, nil, err
    }
    if len(objects) == 0 {
        return 0, nil, errors.New("KEK object not found")
    }

    // Get CKA_ID for wrap/unwrap
    idTemplate := []*pkcs11.Attribute{
        pkcs11.NewAttribute(pkcs11.CKA_ID, nil),
    }
    if err := ctx.GetAttributeValue(session, objects[0], idTemplate); err != nil {
        return 0, nil, err
    }
    idBytes := idTemplate[0].Value()
    if idBytes == nil {
        return 0, nil, errors.New("CKA_ID attribute has no value")
    }

    return objects[0], idBytes, nil
}

// Wrap uses the SoftHSM2 KEK to wrap (encrypt) a DEK.
// Uses CKM_AES_KEY_WRAP_KWP (RFC 3394) if available, otherwise CKM_AES_GCM.
func (m *SoftHSMKeyManager) Wrap(ctx context.Context, dek []byte) ([]byte, int, error) {
    if m.ctx == nil {
        return nil, 0, errors.New("SoftHSM not initialized")
    }

    // Find mechanism: prefer AES Key Wrap with Padding (CKM_AES_KEY_WRAP_KWP)
    mechanism, err := m.ctx.GetMechanismList(m.session)
    if err != nil {
        return nil, 0, err
    }
    var useMechanism *pkcs11.Mechanism
    for _, mech := range mechanism {
        if mech == pkcs11.CKM_AES_KEY_WRAP_KWP {
            useMechanism = pkcs11.NewMechanism(pkcs11.CKM_AES_KEY_WRAP_KWP, nil)
            break
        }
    }
    if useMechanism == nil {
        useMechanism = pkcs11.NewMechanism(pkcs11.CKM_AES_GCM, nil)
    }

    // Wrap the DEK using SoftHSM
    wrapped, err := m.ctx.WrapKey(m.session, *useMechanism, m.kekObject, dek)
    if err != nil {
        return nil, 0, fmt.Errorf("wrap key: %w", err)
    }

    return wrapped, 1, nil
}

// Unwrap uses the SoftHSM2 KEK to unwrap (decrypt) a DEK.
func (m *SoftHSMKeyManager) Unwrap(ctx context.Context, wrappedDEK []byte) ([]byte, error) {
    if m.ctx == nil {
        return nil, errors.New("SoftHSM not initialized")
    }

    // Find mechanism
    mechanism, err := m.ctx.GetMechanismList(m.session)
    if err != nil {
        return nil, err
    }
    var useMechanism *pkcs11.Mechanism
    for _, mech := range mechanism {
        if mech == pkcs11.CKM_AES_KEY_WRAP_KWP {
            useMechanism = pkcs11.NewMechanism(pkcs11.CKM_AES_KEY_WRAP_KWP, nil)
            break
        }
    }
    if useMechanism == nil {
        useMechanism = pkcs11.NewMechanism(pkcs11.CKM_AES_GCM, nil)
    }

    // Template for the unwrapped key
    unwrapTemplate := []*pkcs11.Attribute{
        pkcs11.NewAttribute(pkcs11.CKA_CLASS, pkcs11.CKO_SECRET_KEY),
        pkcs11.NewAttribute(pkcs11.CKA_KEY_TYPE, pkcs11.CKK_AES),
        pkcs11.NewAttribute(pkcs11.CKA_VALUE_LEN, uint(len(wrappedDEK))),
        pkcs11.NewAttribute(pkcs11.CKA_ID, m.keyID),
    }

    dek, err := m.ctx.UnwrapKey(m.session, *useMechanism, m.kekObject, wrappedDEK, unwrapTemplate)
    if err != nil {
        return nil, fmt.Errorf("unwrap key: %w", err)
    }
    if dek == 0 {
        return nil, errors.New("unwrap returned invalid handle")
    }

    // Get the value of the unwrapped key
    valueTemplate := []*pkcs11.Attribute{
        pkcs11.NewAttribute(pkcs11.CKA_VALUE, nil),
    }
    if err := m.ctx.GetAttributeValue(m.session, dek, valueTemplate); err != nil {
        return nil, fmt.Errorf("get unwrapped key value: %w", err)
    }
    val := valueTemplate[0].Value()
    if val == nil {
        return nil, errors.New("CKA_VALUE attribute has no value")
    }
    return val, nil
}

// GetKeyVersion returns 1 for SoftHSM (versioning managed externally via token objects).
func (m *SoftHSMKeyManager) GetKeyVersion(ctx context.Context) (int, error) {
    return 1, nil
}

// Close closes the SoftHSM session and finalizes the context.
func (m *SoftHSMKeyManager) Close() error {
    if m.ctx != nil && m.session != 0 {
        m.ctx.Logout(m.session)
        m.ctx.CloseSession(m.session)
        m.ctx.Finalize()
    }
    return nil
}
```

```go
package crypto

import (
    "context"
    "time"
)

// EncryptedSecret represents an encrypted RADIUS shared secret stored in PostgreSQL.
// Source: docs/design/17_crypto.md §8
type EncryptedSecret struct {
    ID            string    `json:"id"`            // UUID of the secret record
    AaaConfigID  string    `json:"aaa_config_id"` // FK to aaa_server_configs
    Ciphertext   []byte   `json:"ct"`            // DEK-encrypted secret (binary)
    Nonce        []byte   `json:"n"`             // 12-byte nonce (binary)
    Tag          []byte   `json:"t"`             // 16-byte GCM tag (binary)
    EncryptedDEK []byte   `json:"ed"`           // DEK encrypted with KEK (binary)
    DEKVersion  int       `json:"dv"`           // KEK version
    Version      int       `json:"v"`            // Secret version (for rotation)
    CreatedAt    time.Time `json:"created_at"`
    ExpiresAt    time.Time `json:"expires_at"`
    IsActive     bool      `json:"is_active"`
}

// EncryptSecret encrypts a plaintext shared secret for storage.
// Called when configuring a new AAA server or rotating secrets (D-06).
func EncryptSecret(ctx context.Context, plaintextSecret string, km KeyManager) (*EncryptedSecret, error) {
    // Get current KEK version
    ver, err := km.GetKeyVersion(ctx)
    if err != nil {
        return nil, err
    }

    // Generate per-secret DEK
    dek, err := GenerateDEK()
    if err != nil {
        return nil, err
    }

    // Encrypt secret with DEK
    dataEnc, err := Encrypt([]byte(plaintextSecret), dek, nil)
    if err != nil {
        return nil, err
    }

    // Wrap DEK with KEK via KeyManager
    wrappedDEK, _, err := km.Wrap(ctx, dek)
    if err != nil {
        return nil, err
    }

    now := time.Now()
    return &EncryptedSecret{
        Ciphertext:   dataEnc.Ciphertext,
        Nonce:        dataEnc.Nonce,
        Tag:          dataEnc.Tag,
        EncryptedDEK: wrappedDEK,
        DEKVersion:  ver,
        Version:      1,
        CreatedAt:    now,
        ExpiresAt:    now.Add(90 * 24 * time.Hour), // 90-day secret
        IsActive:     true,
    }, nil
}

// DecryptSecret decrypts a stored shared secret.
// Used by RADIUS client to decrypt before computing Message-Authenticator.
func DecryptSecret(ctx context.Context, es *EncryptedSecret, km KeyManager) (string, error) {
    // Unwrap DEK with KEK via KeyManager
    dek, err := km.Unwrap(ctx, es.EncryptedDEK)
    if err != nil {
        return "", err
    }

    // Decrypt secret with DEK
    raw, err := Decrypt(EncryptedData{
        Ciphertext: es.Ciphertext,
        Nonce:     es.Nonce,
        Tag:       es.Tag,
    }, dek, nil)
    if err != nil {
        return "", err
    }

    return string(raw), nil
}

// RotateSecret generates a new encrypted version of a secret.
// The old secret remains valid during the overlap window.
func RotateSecret(ctx context.Context, plaintextSecret string, km KeyManager) (*EncryptedSecret, error) {
    enc, err := EncryptSecret(ctx, plaintextSecret, km)
    if err != nil {
        return nil, err
    }
    enc.Version++
    return enc, nil
}
```

**acceptance_criteria:**
- `EncryptSecret(ctx, "shared_secret", km)` returns `*EncryptedSecret`
- `DecryptSecret(ctx, encrypted, km)` round-trips to original plaintext
- `EncryptedSecret` has fields: `Ciphertext`, `Nonce`, `Tag`, `EncryptedDEK`, `DEKVersion`, `Version`, `CreatedAt`, `ExpiresAt`, `IsActive`
- `EncryptedSecret.ExpiresAt` is 90 days after creation
