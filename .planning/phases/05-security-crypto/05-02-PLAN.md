```go
package crypto

import (
    "context"
    "encoding/json"
    "time"

    "github.com/operator/nssAAF/internal/types"
)

// EncryptedSession represents an encrypted EAP session state stored in PostgreSQL.
// All BYTEA fields are binary (no base64 encoding).
type EncryptedSession struct {
    AuthCtxID    string    `json:"auth_ctx_id"`
    Ciphertext   []byte    `json:"ct"`   // AES-256-GCM encrypted session (binary)
    Nonce        []byte    `json:"n"`    // 12-byte nonce (binary)
    Tag          []byte    `json:"t"`    // 16-byte GCM tag (binary)
    EncryptedDEK []byte    `json:"ed"`   // DEK encrypted with KEK (binary)
    DEKVersion  int       `json:"dv"`   // KEK version used
    EncryptedAt  time.Time `json:"ea"`
}

// SessionState is the plaintext session data that gets encrypted.
// This is a generic JSON-serializable container â€” the actual EAP session
// data is stored inside StateJSON as a serialized proto or JSON blob.
type SessionState struct {
    AuthCtxID  string          `json:"auth_ctx_id"`
    Supi       string          `json:"supi"`
    Gpsi       string          `json:"gpsi"`
    Snssai     *types.Snssai    `json:"snssai,omitempty"`
    StateJSON  json.RawMessage `json:"state_json,omitempty"`
    CreatedAt  time.Time       `json:"created_at"`
    UpdatedAt  time.Time       `json:"updated_at"`
}

// EncryptSession encrypts a session state using a per-session DEK wrapped by KEK.
// The DEK is generated fresh via crypto/rand. The KEK version is fetched from keyManager.
func EncryptSession(ctx context.Context, state *SessionState, km KeyManager) (*EncryptedSession, error) {
    // Serialize state to JSON
    raw, err := json.Marshal(state)
    if err != nil {
        return nil, err
    }

    ver, err := km.GetKeyVersion(ctx)
    if err != nil {
        return nil, err
    }

    // Generate per-session DEK
    dek, err := GenerateDEK()
    if err != nil {
        return nil, err
    }

    // Encrypt data with DEK
    dataEnc, err := Encrypt(raw, dek, nil)
    if err != nil {
        return nil, err
    }

    // Wrap DEK with KEK via KeyManager
    wrappedDEK, _, err := km.Wrap(ctx, dek)
    if err != nil {
        return nil, err
    }

    return &EncryptedSession{
        AuthCtxID:    state.AuthCtxID,
        Ciphertext:   dataEnc.Ciphertext,
        Nonce:        dataEnc.Nonce,
        Tag:          dataEnc.Tag,
        EncryptedDEK: wrappedDEK,
        DEKVersion:   ver,
        EncryptedAt:  time.Now(),
    }, nil
}

// DecryptSession uses KeyManager to unwrap the DEK.
func DecryptSession(ctx context.Context, es *EncryptedSession, km KeyManager) (*SessionState, error) {
    // Unwrap DEK with KEK via KeyManager
    dek, err := km.Unwrap(ctx, es.EncryptedDEK)
    if err != nil {
        return nil, err
    }

    // Decrypt data with DEK
    raw, err := Decrypt(EncryptedData{
        Ciphertext: es.Ciphertext,
        Nonce:     es.Nonce,
        Tag:       es.Tag, // EncryptedSession.Tag = data encryption tag = Envelope.DataTag
    }, dek, nil)
    if err != nil {
        return nil, err
    }

    var state SessionState
    if err := json.Unmarshal(raw, &state); err != nil {
        return nil, err
    }

    return &state, nil
}
```

**Note:** `EncryptedSession` stores only the data encryption tag (`es.Tag` = `Envelope.DataTag`). The DEK tag (`Envelope.DEKTag`) is not stored in `EncryptedSession` â€” the wrapped DEK is already an authenticated ciphertext from the KeyManager. If KeyManager.Wrap returns the wrapped result in an authenticated format, the DEK integrity is protected by that mechanism.
```

```go
package crypto

import (
    "context"
    "sync"
    "time"
)

// KEKRotator manages KEK version lifecycle with a 30-day overlap window.
// Supports zero-downtime rotation: during overlap, both current and previous KEK are valid.
type KEKRotator struct {
    mu          sync.RWMutex
    currentVer   int
    currentExpiry time.Time // when current becomes "previous" (unused if no expiry)
    previousVer  int
    previousExpiry time.Time
    overlapDays  int
    db           interface{} // placeholder for future DB integration
}

// NewKEKRotator creates a KEKRotator with a given overlap window in days.
func NewKEKRotator(overlapDays int) *KEKRotator {
    if overlapDays <= 0 {
        overlapDays = 30
    }
    return &KEKRotator{
        currentVer:  1,
        overlapDays: overlapDays,
    }
}

// GetCurrentVersion returns the current active KEK version.
func (r *KEKRotator) GetCurrentVersion() int {
    r.mu.RLock()
    defer r.mu.RUnlock()
    return r.currentVer
}

// GetPreviousVersion returns the previous KEK version (valid during overlap), or 0 if none.
func (r *KEKRotator) GetPreviousVersion() int {
    r.mu.RLock()
    defer r.mu.RUnlock()
    return r.previousVer
}

// Rotate records a KEK rotation event.
// The new version becomes current; the old current becomes previous.
// After overlapDays, the previous is discarded.
func (r *KEKRotator) Rotate(ctx context.Context) error {
    r.mu.Lock()
    defer r.mu.Unlock()

    r.previousVer = r.currentVer
    r.currentVer++
    r.previousExpiry = time.Now().Add(time.Duration(r.overlapDays) * 24 * time.Hour)

    return nil
}

// IsInOverlapWindow returns true if the previous KEK is still valid.
func (r *KEKRotator) IsInOverlapWindow() bool {
    r.mu.RLock()
    defer r.mu.RUnlock()
    if r.previousVer == 0 {
        return false
    }
    return time.Now().Before(r.previousExpiry)
}

// DiscardPrevious clears the previous KEK version after the overlap window.
// This should be called by a background goroutine or cron job.
func (r *KEKRotator) DiscardPrevious() {
    r.mu.Lock()
    defer r.mu.Unlock()
    if r.previousVer != 0 && time.Now().After(r.previousExpiry) {
        r.previousVer = 0
    }
}

// VersionInfo holds the current and previous KEK version metadata.
type VersionInfo struct {
    Current  int
    Previous int
    OverlapEndsAt *time.Time
}

// GetVersionInfo returns the current and previous KEK version metadata.
func (r *KEKRotator) GetVersionInfo() VersionInfo {
    r.mu.RLock()
    defer r.mu.RUnlock()
    info := VersionInfo{Current: r.currentVer, Previous: r.previousVer}
    if r.previousVer != 0 {
        info.OverlapEndsAt = &r.previousExpiry
    }
    return info
}
```

**acceptance_criteria:**
- `NewKEKRotator(30)` creates a rotator with currentVer=1, overlapDays=30
- `KEKRotator.Rotate()` increments currentVer and sets previousVer to old current
- `KEKRotator.GetCurrentVersion()` returns the current version
- `KEKRotator.GetPreviousVersion()` returns 0 before first rotation
- `KEKRotator.IsInOverlapWindow()` returns true after rotation (until expiry)
- `KEKRotator.DiscardPrevious()` clears previousVer after overlap window
